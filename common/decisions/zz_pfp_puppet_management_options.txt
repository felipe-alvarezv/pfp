pfp_puppet_management = {

	#Decision to show PFP GUI
    pfp_management_window = {
		icon = generic_industry

		available = { 
            is_ai = no
        }

		fire_only_once = no
		complete_effect = { 
            set_country_flag = pfp_management_window
        }
		ai_will_do = { factor = 0 }
	}

	#Decision to use overlord color
	pfp_management_use_overlord_color = {
		icon = generic_industry

		available = {
			custom_trigger_tooltip = {
				tooltip = pfp_management_use_overlord_color_tooltip
				all_country = {
					has_war = no
				}
			}
		}

		fire_only_once = no

		complete_effect = {
			hidden_effect = {
				if = {
					limit = { NOT = { has_global_flag = pfp_use_overlord_color } } #If use overlord color is disabled
					
					set_global_flag = pfp_use_overlord_color #Set the flag to enable use overlord color
					
					every_country = {
						every_subject_country = {
							limit = { has_autonomy_state = pfp_autonomy_military_occupation_painted } #Check if autonomy is not using overlord color
							set_temp_variable = { target_military_occupation = THIS }
							overlord = { set_autonomy = { target = target_military_occupation autonomy_state = pfp_autonomy_military_occupation } }
						}
					}
				}
				else = { #If use overlord color is enabled
					clr_global_flag = pfp_use_overlord_color #Clear the flag to disable use overlord color

					every_country = {
						every_subject_country = {
							limit = { has_autonomy_state = pfp_autonomy_military_occupation } #Check if autonomy is using overlord color
							set_temp_variable = { target_military_occupation = THIS }
							overlord = { set_autonomy = { target = target_military_occupation autonomy_state = pfp_autonomy_military_occupation_painted } }
						}
					}
				}
			}
		}
		ai_will_do = { factor = 0 }
	}

	#Decision to toggle resistance mechanics
	pfp_management_resistance = {
		icon = guerilla_base

		available = {
			is_ai = no
		}

		fire_only_once = no

		complete_effect = {
			hidden_effect = {
				if = {
					limit = { has_global_flag = pfp_management_resistance_enabled }

					clr_global_flag = pfp_management_resistance_enabled
				}
				else = {
					set_global_flag = pfp_management_resistance_enabled
				}
			}
		}
		ai_will_do = { factor = 0 }
	}
	
	#Decision for formable puppet AI to core states with compliance > 80 and resistance < 20
	pfp_management_puppet_core_state = {
		available = {
			custom_trigger_tooltip = {
				tooltip = pfp_management_puppet_core_state_trigger_tooltip
				any_state_in = {
					array = ROOT.pfp_possible_cores
					is_fully_controlled_by = ROOT
					NOT = { is_core_of = ROOT }

					if = {
						limit = { has_global_flag = pfp_management_resistance_enabled }
						compliance > 80
						resistance < 20
					}
				}
			}
		}

		visible = {
			OR = {
				has_autonomy_state = pfp_autonomy_military_occupation
				has_autonomy_state = pfp_autonomy_military_occupation_painted
			}
		}
		
		fire_only_once = no

		complete_effect = {
			custom_effect_tooltip  = pfp_management_puppet_core_state_effect_tooltip
			hidden_effect  = {
				for_each_scope_loop = {
					array = ROOT.pfp_possible_cores

					if = {
						limit = {
							is_fully_controlled_by = ROOT
							NOT = { is_core_of = ROOT }

							if = {
								limit = { has_global_flag = pfp_management_resistance_enabled }
								compliance > 80
								resistance < 20
							}
						}
						add_core_of = ROOT
					}
				}
			}
		}
		ai_will_do = { factor = 1 }
	}
}
