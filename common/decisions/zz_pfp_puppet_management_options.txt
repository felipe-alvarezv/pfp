pfp_puppet_management = {

	#Decision to show PFP GUI
    pfp_management_window = {
		icon = generic_industry

		available = { 
            is_ai = no
        }

		fire_only_once = no
		complete_effect = { 
            set_country_flag = pfp_management_window
        }
		ai_will_do = { factor = 0 }
	}

	#Decision to toggle the color of the "Military Occupation"
	pfp_management_coloured_puppets = {
		icon = generic_industry

		available = {
			is_ai = no
		}

		fire_only_once = no

		complete_effect = {
			hidden_effect = {
				if = {
					limit = { has_global_flag = pfp_autonomy_military_occupation_colour_on } #Check if overlord color is toggled off
					
					clr_global_flag = pfp_autonomy_military_occupation_colour_on
					
					every_country = {
						every_subject_country = {
							limit = { has_autonomy_state = pfp_autonomy_military_occupation_colour }
							set_temp_variable = { target_military_occupation = THIS }
							overlord = { set_autonomy = { target = target_military_occupation autonomy_state = pfp_autonomy_military_occupation } } #Toggle overlord color off
						}
					}
				}
				else = { 
					set_global_flag = pfp_autonomy_military_occupation_colour_on

					every_country = {
						every_subject_country = {
							limit = { has_autonomy_state = pfp_autonomy_military_occupation }
							set_temp_variable = { target_military_occupation = THIS }
							overlord = { set_autonomy = { target = target_military_occupation autonomy_state = pfp_autonomy_military_occupation_colour } } #Toggle overlord color on
						}
					}
				}
			}
		}
		ai_will_do = { factor = 0 }
	}

	#Decision to toggle resistance mechanics
	pfp_management_resistance = {
		icon = guerilla_base

		available = {
			is_ai = no
		}

		fire_only_once = no

		complete_effect = {
			hidden_effect = {
				if = {
					limit = { has_global_flag = pfp_management_resistance_enabled }

					clr_global_flag = pfp_management_resistance_enabled
				}
				else = {
					set_global_flag = pfp_management_resistance_enabled
				}
			}
		}
	}

	#Decision for formable puppet AI to core states with compliance > 80 and resistance < 20
	pfp_management_puppet_core_state = {
		available = {
			#If resistance is enabled
			if = {
				limit = { has_global_flag = pfp_management_resistance_enabled }
				all_controlled_state = {
					is_in_array = {
						array = ROOT.pfp_possible_cores
						value = THIS.id
					}
					NOT = { is_core_of = ROOT }
					compliance > 80
					resistance < 20
				}
			}
			#If resistance is disabled
			else = {
				all_controlled_state = {
					is_in_array = {
						array = ROOT.pfp_possible_cores
						value = THIS.id
					}
					NOT = { is_core_of = ROOT }
				}
			}
		}

		visible = {
			OR = {
				has_autonomy_state = pfp_autonomy_military_occupation
				has_autonomy_state = pfp_autonomy_military_occupation_colour
			}
		}

		fire_only_once = no

		complete_effect = {
			hidden_effect = {
				if = {
					every_controlled_state = {
						limit = {
							is_in_array = {
								array = ROOT.pfp_possible_cores
								value = THIS.id
							}
							compliance > 80
							resistance < 20
						}
						add_core_of = ROOT
					}
				}
				else = {
					every_controlled_state = {
						limit = {
							is_in_array = {
								array = ROOT.pfp_possible_cores
								value = THIS.id
							}
						}
						add_core_of = ROOT
					}
				}
			}
		}
	}
}
