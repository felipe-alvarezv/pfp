pfp_puppet_management = {

	#Decision to show PFP GUI
    pfp_management_window = {
		icon = generic_industry

		available = { 
            is_ai = no
        }

		fire_only_once = no
		complete_effect = { 
            set_country_flag = pfp_management_window
        }
		ai_will_do = { factor = 0 }
	}

	#Decision to toggle the color of the "Military Occupation"
	pfp_management_colored_puppets = {
		icon = generic_industry

		available = {
			is_ai = no
		}

		fire_only_once = no

		complete_effect = {
			hidden_effect = {
				if = {
					limit = { has_global_flag = pfp_use_overlord_color } #Check if overlord color is toggled off
					
					clr_global_flag = pfp_use_overlord_color
					
					every_country = {
						every_subject_country = {
							limit = { has_autonomy_state = pfp_autonomy_military_occupation_color }
							set_temp_variable = { target_military_occupation = THIS }
							overlord = { set_autonomy = { target = target_military_occupation autonomy_state = pfp_autonomy_military_occupation } } #Toggle overlord color off
						}
					}
				}
				else = { 
					set_global_flag = pfp_use_overlord_color

					every_country = {
						every_subject_country = {
							limit = { has_autonomy_state = pfp_autonomy_military_occupation }
							set_temp_variable = { target_military_occupation = THIS }
							overlord = { set_autonomy = { target = target_military_occupation autonomy_state = pfp_autonomy_military_occupation_color } } #Toggle overlord color on
						}
					}
				}
			}
		}
		ai_will_do = { factor = 0 }
	}

	#Decision to toggle resistance mechanics
	pfp_management_resistance = {
		icon = guerilla_base

		available = {
			is_ai = no
		}

		fire_only_once = no

		complete_effect = {
			hidden_effect = {
				if = {
					limit = { has_global_flag = pfp_management_resistance_enabled }

					clr_global_flag = pfp_management_resistance_enabled
				}
				else = {
					set_global_flag = pfp_management_resistance_enabled
				}
			}
		}
	}

	#Decision for formable puppet AI to core states with compliance > 80 and resistance < 20
	pfp_management_puppet_core_state = {
		available = {
			any_state_in = {
				array = ROOT.pfp_possible_cores
				is_fully_controlled_by = ROOT
				NOT = { is_core_of = ROOT }

				if = {
					limit = { has_global_flag = pfp_management_resistance_enabled }
					compliance > 80
					resistance < 20
				}
			}
		}

		visible = {
			OR = {
				has_autonomy_state = pfp_autonomy_military_occupation
				has_autonomy_state = pfp_autonomy_military_occupation_color
			}
		}
		
		fire_only_once = no

		complete_effect = {
			hidden_effect = {
				for_each_scope_loop = {
					array = ROOT.pfp_possible_cores

					if = {
						limit = {
							is_fully_controlled_by = ROOT
							NOT = { is_core_of = ROOT }

							if = {
								limit = { has_global_flag = pfp_management_resistance_enabled }
								compliance > 80
								resistance < 20
							}
						}
						add_core_of = ROOT
					}
				}
			}
		}
	}
}
