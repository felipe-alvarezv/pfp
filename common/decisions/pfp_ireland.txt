pfp_category = {
	pfp_ireland = {
		allowed = {
			NOT = {
				original_tag = IRE #Ireland
				original_tag = NIR #Northern Ireland
			}
		}
		available = {
			is_subject = no
			#Leinster
			custom_trigger_tooltip = {
				tooltip = pfp_ireland_required_states_leinster
				has_full_control_of_state = 113 #Leinster
			}
			#Connaught
			custom_trigger_tooltip = {
				tooltip = pfp_ireland_required_states_connaught
				has_full_control_of_state = 134 #Connaught
			}
			#Munster
			custom_trigger_tooltip = {
				tooltip = pfp_ireland_required_states_munster
				has_full_control_of_state = 135 #Munster
			}
		}
		visible = {
			OR = {
				has_full_control_of_state = 113 #Leinster
				has_full_control_of_state = 134 #Connaught
				has_full_control_of_state = 135 #Munster
			}
			NOT = { has_global_flag = pfp_ireland_flag }
		}
		highlight_states = {
			highlight_state_targets = { 
				state = 113 #Leinster
				state = 134 #Connaught
				state = 135 #Munster
			}
		}
		icon = generic_military
		complete_effect = {
			custom_effect_tooltip = pfp_ireland_core_states
			custom_effect_tooltip = pfp_ireland_transfer_states
			custom_effect_tooltip = pfp_ireland_puppet

			#Create formable puppet with all its states and then puppet
			hidden_effect = {
				#Transfer control and ownership of capital to formable puppet
				113 = { transfer_state_to = P73 } #Leinster

				#Formable puppet becomes a puppet of ROOT
				puppet = P73

				#Code to assign generic portraits and set puppet color
				set_temp_variable = { pfp_portrait_tag = P73 }
				pfp_set_generic_portrait = yes
				pfp_set_autonomy_color = yes

				#Formable puppet will core or contest its designated territory
				pfp_release_ireland = yes

				#Transfer all technology from ROOT to formable puppet
				P73 = { pfp_transfer_technology = yes }

				#Set global flag to prevent future decision use
				set_global_flag = pfp_ireland_flag
			}
		}
		ai_will_do = { factor = 0 }
	}
}
