pfp_category = {
	pfp_colombia = {
		allowed = {
			NOT = {
				original_tag = COL #Colombia
			}
		}
		available = {
			is_subject = no
			#Cundinamarca
			custom_trigger_tooltip = {
				tooltip = pfp_colombia_required_states_cundinamarca
				has_full_control_of_state = 306 #Cundinamarca
			}
			#Meta
			custom_trigger_tooltip = {
				tooltip = pfp_colombia_required_states_meta
				has_full_control_of_state = 486 #Meta
			}
			#La Libertad
			custom_trigger_tooltip = {
				tooltip = pfp_colombia_required_states_la_libertad
				has_full_control_of_state = 493 #La Libertad
			}
		}
		visible = {
			OR = { 
				has_full_control_of_state = 306 #Cundinamarca
				has_full_control_of_state = 486 #Meta
				has_full_control_of_state = 493 #La Libertad
			}
			NOT = { has_global_flag = pfp_colombia_flag }
		}
		highlight_states = {
			highlight_state_targets = { 
				state = 306 #Cundinamarca
				state = 486 #Meta
				state = 493 #La Libertad
			}
		}
		icon = generic_military
		complete_effect = {
			custom_effect_tooltip = pfp_colombia_core_states
			custom_effect_tooltip = pfp_colombia_transfer_states
			custom_effect_tooltip = pfp_colombia_puppet

			#Create formable puppet with all its states and then puppet
			hidden_effect = {
				#Transfer control and ownership of capital to formable puppet
				306 = { transfer_state_to = P01 } #Cundinamarca

				#Formable puppet becomes a puppet of ROOT
				puppet = P01

				#Code to assign generic portraits and set puppet color
				set_temp_variable = { pfp_portrait_tag = P01 }
				pfp_set_generic_portrait = yes
				pfp_set_autonomy_color = yes

				#Formable puppet will core or contest its designated territory
				pfp_release_colombia = yes

				#Transfer all technology from ROOT to formable puppet
				P01 = { pfp_transfer_technology = yes }

				#Set global flag to prevent future decision use
				set_global_flag = pfp_colombia_flag
			}
		}
		ai_will_do = { factor = 0 }
	}
}
