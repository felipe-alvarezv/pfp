pfp_category = {
    pfp_taiwan = {
        allowed = {
            NOT = {
                original_tag = CHI #China
                original_tag = FOR #Taiwan
                original_tag = PRC #Chinese Soviet Republic
            }
        }
        available = {
            is_subject = no
            #Taiwan
            custom_trigger_tooltip = {
                tooltip = pfp_taiwan_required_states_taiwan
                has_full_control_of_state = 524 #Taiwan
            }
        }
        visible = {
            OR = {
                has_full_control_of_state = 524 #Taiwan
            }
            NOT = { has_global_flag = pfp_taiwan_flag }
        }
        highlight_states = {
            highlight_state_targets = {
                state = 524 #Taiwan
            }
        }
        icon = generic_military
        complete_effect = {
            custom_effect_tooltip = pfp_taiwan_core_states
            custom_effect_tooltip = pfp_taiwan_transfer_states
            custom_effect_tooltip = pfp_taiwan_puppet

            #Create formable puppet with all its states and then puppet
            hidden_effect = {
                #Transfer control and ownership of capital to formable puppet
                524 = { transfer_state_to = Q22 } #Taiwan

                #Formable puppet becomes a puppet of ROOT
                puppet = Q22

                #Code to assign generic portraits and set puppet color
                set_temp_variable = { pfp_portrait_tag = Q22 }
                pfp_set_generic_portrait = yes
                pfp_set_autonomy_colour = yes

                #Formable puppet will core or contest its designated territory
                pfp_release_taiwan = yes

                #Transfer all technology from ROOT to formable puppet
                Q22 = { pfp_transfer_technology = yes }

                #Set global flag to prevent future decision use
                set_global_flag = pfp_taiwan_flag
            }
        }
        ai_will_do = { factor = 0 }
    }
}
