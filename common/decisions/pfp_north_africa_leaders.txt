	#Dismiss Leader
	pfp_north_africa_leader_dismiss = {
		allowed = {
			NOT = { original_tag = P30 } #Check if not North Africa
		}
		available = {
			NOT = { original_tag = P30 } #Check if not North Africa
		}
		visible = {
			is_subject = no #FROM must not be a subject
			has_global_flag = pfp_north_africa_flag #Check if North African puppet has been made
			has_global_flag =  pfp_north_africa_leader_flag  #Check if leader was appointed
			country_exists = P30 #Check if North Africa exists
			P30 = {
				is_puppet_of = FROM #Check if North Africa is puppet of FROM
				has_government = FROM #Check if North Africa has same government as FROM
			}
		}
		complete_effect = {
			custom_effect_tooltip = pfp_north_africa_leader_dismiss_information
			hidden_effect = {
				P30 = { kill_country_leader = yes } #Retire the current leader of North Africa

				FROM = {
					if = {
						limit = { has_government = democratic }
						P30 = {
							create_country_leader = {
								name = "Military Junta"
								desc = ""
								picture = "gfx/leaders/PFP/pfp_democratic.dds"
								ideology = conservatism
							}
						}
					}
					else_if = {
						limit = { has_government = communism }
						P30 = {
							create_country_leader = {
								name = "Military Junta"
								desc = ""
								picture = "gfx/leaders/PFP/pfp_communism.dds"
								ideology = marxism
							}
						}
					}
					else_if = {
						limit = { has_government = fascism }
						P30 = {
							create_country_leader = {
								name = "Military Junta"
								desc = ""
								picture = "gfx/leaders/PFP/pfp_fascism.dds"
								ideology = fascism_ideology
							}
						}
					}
					else_if = {
						limit = { has_government = neutrality }
						P30 = {
							create_country_leader = {
								name = "Military Junta"
								desc = ""
								picture = "gfx/leaders/PFP/pfp_neutrality.dds"
								ideology = despotism
							}
						}
					}
				}
				clr_global_flag = pfp_north_africa_leader_flag
			}
		}
		ai_will_do = {
			factor = 0
		}
	}
