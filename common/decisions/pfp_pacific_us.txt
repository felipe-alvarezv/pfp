pfp_category = {
	pfp_pacific_us = {
		allowed = {
			NOT = {
				original_tag = CSA #Confederate States
				original_tag = USA #United States
			}
		}
		available = {
			is_subject = no
			#California
			custom_trigger_tooltip = {
				tooltip = pfp_pacific_us_required_states_california
				has_full_control_of_state = 378 #California
			}
			#Oregon
			custom_trigger_tooltip = {
				tooltip = pfp_pacific_us_required_states_oregon
				has_full_control_of_state = 385 #Oregon
			}
			#Washington
			custom_trigger_tooltip = {
				tooltip = pfp_pacific_us_required_states_washington
				has_full_control_of_state = 386 #Washington
			}
		}
		visible = {
			OR = {
				has_full_control_of_state = 378 #California
				has_full_control_of_state = 385 #Oregon
				has_full_control_of_state = 386 #Washington
			}
			NOT = { has_global_flag = pfp_pacific_us_flag }
		}
		highlight_states = {
			highlight_state_targets = { 
				state = 378 #California
				state = 385 #Oregon
				state = 386 #Washington
			}
		}
		icon = generic_military
		complete_effect = {
			custom_effect_tooltip = pfp_pacific_us_core_states
			custom_effect_tooltip = pfp_pacific_us_transfer_states
			custom_effect_tooltip = pfp_pacific_us_puppet

			#Create formable puppet with all its states and then puppet
			hidden_effect = {
				#Transfer control and ownership of capital to formable puppet
				378 = { transfer_state_to = P21 } #California

				#Formable puppet becomes a puppet of ROOT
				puppet = P21

				#Code to assign generic portraits and set puppet color
				set_temp_variable = { pfp_portrait_tag = P21 }
				pfp_set_generic_portrait = yes
				pfp_set_autonomy_colour = yes

				#Formable puppet will core or contest its designated territory
				pfp_release_pacific_us = yes

				#Transfer all technology from ROOT to formable puppet
				P21 = { pfp_transfer_technology = yes }

				#Set global flag to prevent future decision use
				set_global_flag = pfp_pacific_us_flag
			}
		}
		ai_will_do = { factor = 0 }
	}
}
