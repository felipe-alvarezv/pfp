#Scripted effect to release Central Russia (P39)
pfp_release_central_russia = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_enabled }

			#Add constestion to formable puppet
			205 = { add_contested_owner  = P39 } #Kaluga
			219 = { add_contested_owner  = P39 } #Moscow
			220 = { add_contested_owner  = P39 } #Kursk
			222 = { add_contested_owner  = P39 } #Orel
			223 = { add_contested_owner  = P39 } #Tula
			224 = { add_contested_owner  = P39 } #Bryansk
			240 = { add_contested_owner  = P39 } #Belgorod
			246 = { add_contested_owner  = P39 } #Rzhev
			247 = { add_contested_owner  = P39 } #Kalinin
			248 = { add_contested_owner  = P39 } #Yaroslavl
			253 = { add_contested_owner  = P39 } #Ivanovo
			254 = { add_contested_owner  = P39 } #Ryazan
			257 = { add_contested_owner  = P39 } #Tambov
			258 = { add_contested_owner  = P39 } #Lipetsk
			260 = { add_contested_owner  = P39 } #Voronezh

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P39 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P39
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P39
				}
			}

			#Set custom focus tree to combat resistance
			P39 = { load_focus_tree = pfp_formable_puppet_focus }
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			205 = { add_core_of = P39 } #Kaluga
			219 = { add_core_of = P39 } #Moscow
			220 = { add_core_of = P39 } #Kursk
			222 = { add_core_of = P39 } #Orel
			223 = { add_core_of = P39 } #Tula
			224 = { add_core_of = P39 } #Bryansk
			240 = { add_core_of = P39 } #Belgorod
			246 = { add_core_of = P39 } #Rzhev
			247 = { add_core_of = P39 } #Kalinin
			248 = { add_core_of = P39 } #Yaroslavl
			253 = { add_core_of = P39 } #Ivanovo
			254 = { add_core_of = P39 } #Ryazan
			257 = { add_core_of = P39 } #Tambov
			258 = { add_core_of = P39 } #Lipetsk
			260 = { add_core_of = P39 } #Voronezh

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P39 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P39
					transfer_state_to = P39
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P39
					set_state_controller_to = P39
				}
			}
		}	}
}
