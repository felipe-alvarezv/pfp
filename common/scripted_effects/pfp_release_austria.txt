#Scripted effect to release Austria (P77)
pfp_release_austria = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_on }

			#Add constestion to formable puppet
			4 = { add_contested_owner  = P77 } #Lower Austria
			152 = { add_contested_owner  = P77 } #Upper Austria
			153 = { add_contested_owner  = P77 } #Tyrol
			848 = { add_contested_owner  = P77 } #Vorarlberg
			975 = { add_contested_owner  = P77 } #Burgenland
			976 = { add_contested_owner  = P77 } #Steiermark-Kärnten

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P77 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P77
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P77
				}
			}
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			4 = { add_core_of = P77 } #Lower Austria
			152 = { add_core_of = P77 } #Upper Austria
			153 = { add_core_of = P77 } #Tyrol
			848 = { add_core_of = P77 } #Vorarlberg
			975 = { add_core_of = P77 } #Burgenland
			976 = { add_core_of = P77 } #Steiermark-Kärnten

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P77 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P77
					transfer_state_to = P77
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P77
					set_state_controller_to = P77
				}
			}
		}	}
}
