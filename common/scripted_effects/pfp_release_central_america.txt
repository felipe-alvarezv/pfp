#Scripted effect to release Central America (P08)
pfp_release_central_america = {
	hidden_effect = {
		#Add claims to Central America
		304 = { add_claim_by = P08 } #Panama
		312 = { add_claim_by = P08 } #Honduras
		313 = { add_claim_by = P08 } #Guatemala
		314 = { add_claim_by = P08 } #El Salvador
		316 = { add_claim_by = P08 } #Costa Rica
		317 = { add_claim_by = P08 } #Nicaragua

		#Set formable puppet as owner and controller of all its claimed states
		every_state = {
			limit = { is_claimed_by = P08 }
			transfer_state_to = P08
		}

		#Transfer all technology to formable puppet
		P08 = { transfer_technology = yes }

		#Set global flag to prevent future decision use
		set_global_flag = pfp_central_america_flag

		#Send event to all nations about the formable puppet
		set_temp_variable = { temp_country_tag = P08 }
	}
}
