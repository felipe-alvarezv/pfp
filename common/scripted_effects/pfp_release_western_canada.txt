#Scripted effect to release Western Canada (P29)
pfp_release_western_canada = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_enabled }

			#Add constestion to formable puppet
			467 = { add_contested_owner  = P29 } #Manitoba
			469 = { add_contested_owner  = P29 } #Saskatchewan
			470 = { add_contested_owner  = P29 } #Alberta
			472 = { add_contested_owner  = P29 } #Northwest Territories
			683 = { add_contested_owner  = P29 } #Nunavut
			865 = { add_contested_owner  = P29 } #Northern Saskatchewan
			867 = { add_contested_owner  = P29 } #Northern Manitoba

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P29 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P29
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P29
				}
			}
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			467 = { add_core_of = P29 } #Manitoba
			469 = { add_core_of = P29 } #Saskatchewan
			470 = { add_core_of = P29 } #Alberta
			472 = { add_core_of = P29 } #Northwest Territories
			683 = { add_core_of = P29 } #Nunavut
			865 = { add_core_of = P29 } #Northern Saskatchewan
			867 = { add_core_of = P29 } #Northern Manitoba

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P29 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P29
					transfer_state_to = P29
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P29
					set_state_controller_to = P29
				}
			}
		}	}
}
