#Scripted effect to release Kansai (Q04)
pfp_release_kansai = {
    hidden_effect = {
        #Add cores to Kansai
        531 = { add_core_of = Q04 } #Kansai
        
        #Set formable puppet as owner and controller of all its cored states
        every_state = {
            limit = { is_core_of = Q04 }
            add_contested_owner = Q04
            transfer_state_to = Q04
        }

        #Transfer all technology to formable puppet
        Q04 = { pfp_transfer_technology = yes }

        #Set global flag to prevent future decision use
        set_global_flag = pfp_kansai_flag

        #Send event to all nations about the formable puppet
        set_temp_variable = { temp_country_tag = Q04 }
    }
}
