#Scripted effect to release Transcaucasia (P28)
pfp_release_transcaucasia = {
	hidden_effect = {
		#Add claims to Transcaucasia
		229 = { add_claim_by = P28 } #Azerbaijan
		230 = { add_claim_by = P28 } #Armenia
		231 = { add_claim_by = P28 } #Georgia
		826 = { add_claim_by = P28 } #Abkhazia

		#Set formable puppet as owner and controller of all its claimed states
		every_state = {
			limit = { is_claimed_by = P28 }
			transfer_state_to = P28
		}

		#Transfer all technology to formable puppet
		P28 = { transfer_technology = yes }

		#Set global flag to prevent future decision use
		set_global_flag = pfp_transcaucasia_flag

		#Send event to all nations about the formable puppet
		set_temp_variable = { temp_country_tag = P28 }
	}
}
