#Scripted effect to release Greenland (P85)
pfp_release_greenland = {
    hidden_effect = {
        #Add cores to Greenland
        101 = { add_core_of = P85 } #Greenland
        
        #Set formable puppet as owner and controller of all its cored states
        every_state = {
            limit = { is_core_of = P85 }
            add_contested_owner = P85
            transfer_state_to = P85
        }

        #Transfer all technology to formable puppet
        P85 = { pfp_transfer_technology = yes }

        #Set global flag to prevent future decision use
        set_global_flag = pfp_greenland_flag

        #Send event to all nations about the formable puppet
        set_temp_variable = { temp_country_tag = P85 }
    }
}
