#Scripted effect to release Sardinia (P94)
pfp_release_sardinia = {
    hidden_effect = {
        #Add cores to Sardinia
        114 = { add_core_of = P94 } #Sardinia
        
        #Set formable puppet as owner and controller of all its cored states
        every_state = {
            limit = { is_core_of = P94 }
            add_contested_owner = P94
            transfer_state_to = P94
        }

        #Transfer all technology to formable puppet
        P94 = { pfp_transfer_technology = yes }

        #Set global flag to prevent future decision use
        set_global_flag = pfp_sardinia_flag

        #Send event to all nations about the formable puppet
        set_temp_variable = { temp_country_tag = P94 }
    }
}
