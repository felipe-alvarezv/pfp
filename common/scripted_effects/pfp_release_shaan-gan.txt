#Scripted effect to release Shaan-Gan (Q20)
pfp_release_shaan-gan = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_on }

			#Add constestion to formable puppet
			283 = { add_contested_owner  = Q20 } #Gansu
			622 = { add_contested_owner  = Q20 } #Shaanbei
			744 = { add_contested_owner  = Q20 } #Xian
			756 = { add_contested_owner  = Q20 } #Jiuquan
			1032 = { add_contested_owner  = Q20 } #Yan'an
			1044 = { add_contested_owner  = Q20 } #Wuwei
			1045 = { add_contested_owner  = Q20 } #Guyuan
			1046 = { add_contested_owner  = Q20 } #Yulin

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = Q20 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = Q20
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = Q20
				}
			}
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			283 = { add_core_of = Q20 } #Gansu
			622 = { add_core_of = Q20 } #Shaanbei
			744 = { add_core_of = Q20 } #Xian
			756 = { add_core_of = Q20 } #Jiuquan
			1032 = { add_core_of = Q20 } #Yan'an
			1044 = { add_core_of = Q20 } #Wuwei
			1045 = { add_core_of = Q20 } #Guyuan
			1046 = { add_core_of = Q20 } #Yulin

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = Q20 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = Q20
					transfer_state_to = Q20
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = Q20
					set_state_controller_to = Q20
				}
			}
		}

		#Transfer all technology from ROOT to formable puppet
		Q20 = { pfp_transfer_technology = yes }

		#Set global flag to prevent future decision use
		set_global_flag = pfp_shaan-gan_flag

		#Send event to all nations about the formable puppet
		set_temp_variable = { temp_country_tag = Q20 }
	}
}
