#Scripted effect to release Sweden (P74)
pfp_release_sweden = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_on }

			#Add constestion to formable puppet
			38 = { add_contested_owner  = P74 } #Gavleborg
			138 = { add_contested_owner  = P74 } #Skane
			139 = { add_contested_owner  = P74 } #Smaland
			140 = { add_contested_owner  = P74 } #Vastergotland
			141 = { add_contested_owner  = P74 } #Sodermanland
			666 = { add_contested_owner  = P74 } #Norbotten
			913 = { add_contested_owner  = P74 } #Ostergotland
			915 = { add_contested_owner  = P74 } #Bohuslan
			916 = { add_contested_owner  = P74 } #Dalarna
			917 = { add_contested_owner  = P74 } #Jamtland
			918 = { add_contested_owner  = P74 } #Vasterbotten
			919 = { add_contested_owner  = P74 } #Varmland

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P74 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P74
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P74
				}
			}
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			38 = { add_core_of = P74 } #Gavleborg
			138 = { add_core_of = P74 } #Skane
			139 = { add_core_of = P74 } #Smaland
			140 = { add_core_of = P74 } #Vastergotland
			141 = { add_core_of = P74 } #Sodermanland
			666 = { add_core_of = P74 } #Norbotten
			913 = { add_core_of = P74 } #Ostergotland
			915 = { add_core_of = P74 } #Bohuslan
			916 = { add_core_of = P74 } #Dalarna
			917 = { add_core_of = P74 } #Jamtland
			918 = { add_core_of = P74 } #Vasterbotten
			919 = { add_core_of = P74 } #Varmland

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P74 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P74
					transfer_state_to = P74
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P74
					set_state_controller_to = P74
				}
			}
		}	}
}
