#Scripted effect to release Liangjiang (Q14)
pfp_release_liangjiang = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_on }

			#Add constestion to formable puppet
			598 = { add_contested_owner  = Q14 } #Jiangsu
			600 = { add_contested_owner  = Q14 } #Jiangxi
			606 = { add_contested_owner  = Q14 } #Anhui
			613 = { add_contested_owner  = Q14 } #Shanghai
			749 = { add_contested_owner  = Q14 } #Huangshan
			1034 = { add_contested_owner  = Q14 } #Suzhou
			1035 = { add_contested_owner  = Q14 } #Nanjing

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = Q14 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = Q14
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = Q14
				}
			}
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			598 = { add_core_of = Q14 } #Jiangsu
			600 = { add_core_of = Q14 } #Jiangxi
			606 = { add_core_of = Q14 } #Anhui
			613 = { add_core_of = Q14 } #Shanghai
			749 = { add_core_of = Q14 } #Huangshan
			1034 = { add_core_of = Q14 } #Suzhou
			1035 = { add_core_of = Q14 } #Nanjing

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = Q14 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = Q14
					transfer_state_to = Q14
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = Q14
					set_state_controller_to = Q14
				}
			}
		}

		#Transfer all technology to formable puppet
		Q14 = { pfp_transfer_technology = yes }

		#Set global flag to prevent future decision use
		set_global_flag = pfp_liangjiang_flag

		#Send event to all nations about the formable puppet
		set_temp_variable = { temp_country_tag = Q14 }
	}
}
