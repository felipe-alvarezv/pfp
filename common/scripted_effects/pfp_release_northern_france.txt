#Scripted effect to release Northern France (P62)
pfp_release_northern_france = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_on }

			#Add constestion to formable puppet
			16 = { add_contested_owner  = P62 } #Ile de France
			17 = { add_contested_owner  = P62 } #Franche-Comte
			18 = { add_contested_owner  = P62 } #Champagne
			24 = { add_contested_owner  = P62 } #Centre
			27 = { add_contested_owner  = P62 } #Bourgogne
			785 = { add_contested_owner  = P62 } #Picardy

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P62 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P62
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P62
				}
			}
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			16 = { add_core_of = P62 } #Ile de France
			17 = { add_core_of = P62 } #Franche-Comte
			18 = { add_core_of = P62 } #Champagne
			24 = { add_core_of = P62 } #Centre
			27 = { add_core_of = P62 } #Bourgogne
			785 = { add_core_of = P62 } #Picardy

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P62 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P62
					transfer_state_to = P62
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P62
					set_state_controller_to = P62
				}
			}
		}

		#Transfer all technology from ROOT to formable puppet
		P62 = { pfp_transfer_technology = yes }

		#Set global flag to prevent future decision use
		set_global_flag = pfp_northern_france_flag

		#Send event to all nations about the formable puppet
		set_temp_variable = { temp_country_tag = P62 }
	}
}
