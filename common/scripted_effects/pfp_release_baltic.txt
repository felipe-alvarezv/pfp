#Scripted effect to release Baltic (P04)
pfp_release_baltic = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_enabled }

			#Add constestion to formable puppet
			12 = { add_contested_owner  = P04 } #Vidzeme
			13 = { add_contested_owner  = P04 } #Parnu
			190 = { add_contested_owner  = P04 } #Kurzeme
			191 = { add_contested_owner  = P04 } #Tartu
			808 = { add_contested_owner  = P04 } #Riga
			809 = { add_contested_owner  = P04 } #Zemgale
			810 = { add_contested_owner  = P04 } #Latgale
			811 = { add_contested_owner  = P04 } #Saaremaa
			812 = { add_contested_owner  = P04 } #Harju
			813 = { add_contested_owner  = P04 } #Virumaa

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P04 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P04
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P04
				}
			}

			#Set custom focus tree to combat resistance
			P04 = { set_focus_tree = pfp_formable_puppet_focus }
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			12 = { add_core_of = P04 } #Vidzeme
			13 = { add_core_of = P04 } #Parnu
			190 = { add_core_of = P04 } #Kurzeme
			191 = { add_core_of = P04 } #Tartu
			808 = { add_core_of = P04 } #Riga
			809 = { add_core_of = P04 } #Zemgale
			810 = { add_core_of = P04 } #Latgale
			811 = { add_core_of = P04 } #Saaremaa
			812 = { add_core_of = P04 } #Harju
			813 = { add_core_of = P04 } #Virumaa

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P04 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P04
					transfer_state_to = P04
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P04
					set_state_controller_to = P04
				}
			}
		}	}
}
