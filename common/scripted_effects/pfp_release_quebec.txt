#Scripted effect to release Quebec (P22)
pfp_release_quebec = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_enabled }

			#Add constestion to formable puppet
			466 = { add_contested_owner  = P22 } #Nord du Quebec
			468 = { add_contested_owner  = P22 } #Saint Lawrence
			860 = { add_contested_owner  = P22 } #Cote-Nord
			861 = { add_contested_owner  = P22 } #Saguenay
			862 = { add_contested_owner  = P22 } #Ouest du Quebec
			863 = { add_contested_owner  = P22 } #Maurice

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P22 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P22
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P22
				}
			}
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			466 = { add_core_of = P22 } #Nord du Quebec
			468 = { add_core_of = P22 } #Saint Lawrence
			860 = { add_core_of = P22 } #Cote-Nord
			861 = { add_core_of = P22 } #Saguenay
			862 = { add_core_of = P22 } #Ouest du Quebec
			863 = { add_core_of = P22 } #Maurice

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P22 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P22
					transfer_state_to = P22
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P22
					set_state_controller_to = P22
				}
			}
		}	}
}
