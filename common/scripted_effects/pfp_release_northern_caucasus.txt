#Scripted effect to release Northern Caucasus (P41)
pfp_release_northern_caucasus = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_on }

			#Add constestion to formable puppet
			232 = { add_contested_owner  = P41 } #Dagestan
			233 = { add_contested_owner  = P41 } #Sochi
			235 = { add_contested_owner  = P41 } #Stavropol
			821 = { add_contested_owner  = P41 } #Chechnya-Ingushetia
			827 = { add_contested_owner  = P41 } #Kabardino-Balkaria
			828 = { add_contested_owner  = P41 } #North Ossetia

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P41 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P41
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P41
				}
			}
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			232 = { add_core_of = P41 } #Dagestan
			233 = { add_core_of = P41 } #Sochi
			235 = { add_core_of = P41 } #Stavropol
			821 = { add_core_of = P41 } #Chechnya-Ingushetia
			827 = { add_core_of = P41 } #Kabardino-Balkaria
			828 = { add_core_of = P41 } #North Ossetia

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P41 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P41
					transfer_state_to = P41
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P41
					set_state_controller_to = P41
				}
			}
		}

		#Transfer all technology from ROOT to formable puppet
		P41 = { pfp_transfer_technology = yes }

		#Set global flag to prevent future decision use
		set_global_flag = pfp_northern_caucasus_flag

		#Send event to all nations about the formable puppet
		set_temp_variable = { temp_country_tag = P41 }
	}
}
