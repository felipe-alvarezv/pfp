#Scripted effect to release Finland (P12)
pfp_release_finland = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_on }

			#Add constestion to formable puppet
			111 = { add_contested_owner  = P12 } #Uusimaa
			148 = { add_contested_owner  = P12 } #Lappi
			149 = { add_contested_owner  = P12 } #Vaasa
			150 = { add_contested_owner  = P12 } #Kuopio
			926 = { add_contested_owner  = P12 } #Turku
			927 = { add_contested_owner  = P12 } #Häme
			928 = { add_contested_owner  = P12 } #Kymi
			929 = { add_contested_owner  = P12 } #Oulu
			930 = { add_contested_owner  = P12 } #Mikkeli

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P12 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P12
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P12
				}
			}
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			111 = { add_core_of = P12 } #Uusimaa
			148 = { add_core_of = P12 } #Lappi
			149 = { add_core_of = P12 } #Vaasa
			150 = { add_core_of = P12 } #Kuopio
			926 = { add_core_of = P12 } #Turku
			927 = { add_core_of = P12 } #Häme
			928 = { add_core_of = P12 } #Kymi
			929 = { add_core_of = P12 } #Oulu
			930 = { add_core_of = P12 } #Mikkeli

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P12 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P12
					transfer_state_to = P12
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P12
					set_state_controller_to = P12
				}
			}
		}

		#Transfer all technology to formable puppet
		P12 = { pfp_transfer_technology = yes }

		#Set global flag to prevent future decision use
		set_global_flag = pfp_finland_flag

		#Send event to all nations about the formable puppet
		set_temp_variable = { temp_country_tag = P12 }
	}
}
