#Scripted effect to release East Turkestan (Q23)
pfp_release_east_turkestan = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_on }

			#Add constestion to formable puppet
			287 = { add_contested_owner  = Q23 } #Taklamakan
			617 = { add_contested_owner  = Q23 } #Urumqi
			618 = { add_contested_owner  = Q23 } #Dzungaria
			619 = { add_contested_owner  = Q23 } #Yarkand
			759 = { add_contested_owner  = Q23 } #Kunlun Shan
			760 = { add_contested_owner  = Q23 } #Dabacheng
			1042 = { add_contested_owner  = Q23 } #Khotan

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = Q23 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = Q23
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = Q23
				}
			}
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			287 = { add_core_of = Q23 } #Taklamakan
			617 = { add_core_of = Q23 } #Urumqi
			618 = { add_core_of = Q23 } #Dzungaria
			619 = { add_core_of = Q23 } #Yarkand
			759 = { add_core_of = Q23 } #Kunlun Shan
			760 = { add_core_of = Q23 } #Dabacheng
			1042 = { add_core_of = Q23 } #Khotan

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = Q23 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = Q23
					transfer_state_to = Q23
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = Q23
					set_state_controller_to = Q23
				}
			}
		}	}
}
