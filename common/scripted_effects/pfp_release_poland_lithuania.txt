#Scripted effect to release Poland-Lithuania (P34)
pfp_release_poland_lithuania = {
	hidden_effect = {
		#Add cores to Polish-Lithuanian territory
		10 = { add_core_of = P34 } #Warszawa
		11 = { add_core_of = P34 } #Kaunas
		87 = { add_core_of = P34 } #Lodz
		90 = { add_core_of = P34 } #Kielce
		92 = { add_core_of = P34 } #Lublin
		97 = { add_core_of = P34 } #Białystok
		98 = { add_core_of = P34 } #Płock
		189 = { add_core_of = P34 } #Zemaitija
		784 = { add_core_of = P34 } #Wilno
		814 = { add_core_of = P34 } #Suduva
		815 = { add_core_of = P34 } #Aukstaitija

		#Set formable puppet as owner and controller of all its cored states
		every_state = {
			limit = { is_core_of = P34 }
			transfer_state_to = P34
		}

		#Transfer all technology to formable puppet
		P34 = { transfer_technology = yes }

		#Set global flag to prevent future decision use
		set_global_flag = pfp_poland_lithuania_flag

		#Send event to all nations about the formable puppet
		set_temp_variable = { temp_country_tag = P34 }
		news_event = { id = pfp_news.01 }
	}
}
