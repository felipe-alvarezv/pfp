#Scripted effect to release Mittelafrika (P53)
pfp_release_mittelafrika = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_on }

			#Add constestion to formable puppet
			541 = { add_contested_owner  = P53 } #Khomas
			546 = { add_contested_owner  = P53 } #Tanganyika
			773 = { add_contested_owner  = P53 } #Cameroon
			894 = { add_contested_owner  = P53 } #Kuene
			895 = { add_contested_owner  = P53 } #Otjozondjupa
			893 = { add_contested_owner  = P53 } #Karas

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P53 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P53
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P53
				}
			}
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			541 = { add_core_of = P53 } #Khomas
			546 = { add_core_of = P53 } #Tanganyika
			773 = { add_core_of = P53 } #Cameroon
			894 = { add_core_of = P53 } #Kuene
			895 = { add_core_of = P53 } #Otjozondjupa
			893 = { add_core_of = P53 } #Karas

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P53 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P53
					transfer_state_to = P53
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P53
					set_state_controller_to = P53
				}
			}
		}

		#Transfer all technology to formable puppet
		P53 = { pfp_transfer_technology = yes }

		#Set global flag to prevent future decision use
		set_global_flag = pfp_mittelafrika_flag

		#Send event to all nations about the formable puppet
		set_temp_variable = { temp_country_tag = P53 }
	}
}
