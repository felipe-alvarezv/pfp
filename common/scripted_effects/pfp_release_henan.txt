#Scripted effect to release Henan (Q12)
pfp_release_henan = {
    hidden_effect = {
        #Add cores to Henan
        607 = { add_core_of = Q12 } #Henan
        
        #Set formable puppet as owner and controller of all its cored states
        every_state = {
            limit = { is_core_of = Q12 }
            add_contested_owner = Q12
            transfer_state_to = Q12
        }

        #Transfer all technology to formable puppet
        Q12 = { pfp_transfer_technology = yes }

        #Set global flag to prevent future decision use
        set_global_flag = pfp_henan_flag

        #Send event to all nations about the formable puppet
        set_temp_variable = { temp_country_tag = Q12 }
    }
}
