#Scripted effect to release Zhili (Q10)
pfp_release_zhili = {
    hidden_effect = {
        #Add cores to Zhili
        608 = { add_core_of = Q10 } #Beiping
        614 = { add_core_of = Q10 } #Hebei
        1039 = { add_core_of = Q10 } #Hebei-Chahar
        
        #Set formable puppet as owner and controller of all its cored states
        every_state = {
            limit = { is_core_of = Q10 }
            add_contested_owner = Q10
            transfer_state_to = Q10
        }

        #Transfer all technology to formable puppet
        Q10 = { pfp_transfer_technology = yes }

        #Set global flag to prevent future decision use
        set_global_flag = pfp_zhili_flag

        #Send event to all nations about the formable puppet
        set_temp_variable = { temp_country_tag = Q10 }
    }
}
