#Scripted effect to release East Africa (P10)
pfp_release_east_africa = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_on }

			#Add constestion to formable puppet
			271 = { add_contested_owner  = P10 } #Shewa
			550 = { add_contested_owner  = P10 } #Eritrea
			559 = { add_contested_owner  = P10 } #Somaliland
			835 = { add_contested_owner  = P10 } #Hararghe
			836 = { add_contested_owner  = P10 } #Bale
			837 = { add_contested_owner  = P10 } #Sidamo
			838 = { add_contested_owner  = P10 } #Illubabor-Kaffa
			839 = { add_contested_owner  = P10 } #Welega
			840 = { add_contested_owner  = P10 } #Gojjam
			841 = { add_contested_owner  = P10 } #Begemder
			842 = { add_contested_owner  = P10 } #Tigray
			843 = { add_contested_owner  = P10 } #Wello
			844 = { add_contested_owner  = P10 } #Jubaland
			908 = { add_contested_owner  = P10 } #Afar

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P10 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P10
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P10
				}
			}
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			271 = { add_core_of = P10 } #Shewa
			550 = { add_core_of = P10 } #Eritrea
			559 = { add_core_of = P10 } #Somaliland
			835 = { add_core_of = P10 } #Hararghe
			836 = { add_core_of = P10 } #Bale
			837 = { add_core_of = P10 } #Sidamo
			838 = { add_core_of = P10 } #Illubabor-Kaffa
			839 = { add_core_of = P10 } #Welega
			840 = { add_core_of = P10 } #Gojjam
			841 = { add_core_of = P10 } #Begemder
			842 = { add_core_of = P10 } #Tigray
			843 = { add_core_of = P10 } #Wello
			844 = { add_core_of = P10 } #Jubaland
			908 = { add_core_of = P10 } #Afar

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P10 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P10
					transfer_state_to = P10
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P10
					set_state_controller_to = P10
				}
			}
		}

		#Transfer all technology from ROOT to formable puppet
		P10 = { pfp_transfer_technology = yes }

		#Set global flag to prevent future decision use
		set_global_flag = pfp_east_africa_flag

		#Send event to all nations about the formable puppet
		set_temp_variable = { temp_country_tag = P10 }
	}
}
