#Scripted effect to release Southern Russia (P40)
pfp_release_southern_russia = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_on }

			#Add constestion to formable puppet
			217 = { add_contested_owner  = P40 } #Stalingrad
			218 = { add_contested_owner  = P40 } #Rostov
			236 = { add_contested_owner  = P40 } #Astrakhan
			237 = { add_contested_owner  = P40 } #Kalmykia
			238 = { add_contested_owner  = P40 } #Volgodonsk
			245 = { add_contested_owner  = P40 } #Millerovo
			265 = { add_contested_owner  = P40 } #Mikhaylovka

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P40 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P40
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P40
				}
			}
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			217 = { add_core_of = P40 } #Stalingrad
			218 = { add_core_of = P40 } #Rostov
			236 = { add_core_of = P40 } #Astrakhan
			237 = { add_core_of = P40 } #Kalmykia
			238 = { add_core_of = P40 } #Volgodonsk
			245 = { add_core_of = P40 } #Millerovo
			265 = { add_core_of = P40 } #Mikhaylovka

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P40 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P40
					transfer_state_to = P40
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P40
					set_state_controller_to = P40
				}
			}
		}

		#Transfer all technology from ROOT to formable puppet
		P40 = { pfp_transfer_technology = yes }

		#Set global flag to prevent future decision use
		set_global_flag = pfp_southern_russia_flag

		#Send event to all nations about the formable puppet
		set_temp_variable = { temp_country_tag = P40 }
	}
}
