#Scripted effect to release Switzerland (P78)
pfp_release_switzerland = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_enabled }

			#Add constestion to formable puppet
			3 = { add_contested_owner  = P78 } #Swiss Plateau
			151 = { add_contested_owner  = P78 } #Eastern Swiss Alps
			845 = { add_contested_owner  = P78 } #Jura Mountains
			846 = { add_contested_owner  = P78 } #Ticino
			847 = { add_contested_owner  = P78 } #Western Swiss Alps

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P78 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P78
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P78
				}
			}
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			3 = { add_core_of = P78 } #Swiss Plateau
			151 = { add_core_of = P78 } #Eastern Swiss Alps
			845 = { add_core_of = P78 } #Jura Mountains
			846 = { add_core_of = P78 } #Ticino
			847 = { add_core_of = P78 } #Western Swiss Alps

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P78 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P78
					transfer_state_to = P78
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P78
					set_state_controller_to = P78
				}
			}
		}	}
}
