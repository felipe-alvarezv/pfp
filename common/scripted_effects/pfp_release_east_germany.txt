#Scripted effect to release East Germany (P32)
pfp_release_east_germany = {
	hidden_effect = {
		#Add claims to East Germany
		60 = { add_claim_by = P32 } #Thuringen
		61 = { add_claim_by = P32 } #Mecklenburg
		62 = { add_claim_by = P32 } #Vorpommern
		64 = { add_claim_by = P32 } #Brandenburg
		65 = { add_claim_by = P32 } #Sachsen

		#Set formable puppet as owner and controller of all its claimed states
		every_state = {
			limit = { is_claimed_by = P32 }
			transfer_state_to = P32
		}

		#Transfer all technology to formable puppet
		P32 = { transfer_technology = yes }

		#Set global flag to prevent future decision use
		set_global_flag = pfp_east_germany_flag

		#Send event to all nations about the formable puppet
		set_temp_variable = { temp_country_tag = P32 }
	}
}
