#Scripted effect to release Yun-Gui (Q18)
pfp_release_yun-gui = {
    hidden_effect = {
        #Add cores to Yun-Gui
        325 = { add_core_of = Q18 } #Yunnan
        603 = { add_core_of = Q18 } #Guizhou
        747 = { add_core_of = Q18 } #Dali
        748 = { add_core_of = Q18 } #Zunyi
        
        #Set formable puppet as owner and controller of all its cored states
        every_state = {
            limit = { is_core_of = Q18 }
            add_contested_owner = Q18
            transfer_state_to = Q18
        }

        #Transfer all technology to formable puppet
        Q18 = { pfp_transfer_technology = yes }

        #Set global flag to prevent future decision use
        set_global_flag = pfp_yun-gui_flag

        #Send event to all nations about the formable puppet
        set_temp_variable = { temp_country_tag = Q18 }
    }
}
