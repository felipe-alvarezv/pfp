#Scripted effect to release West Africa (P35)
pfp_release_west_africa = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_on }

			#Add constestion to formable puppet
			272 = { add_contested_owner  = P35 } #Senegal
			515 = { add_contested_owner  = P35 } #Southern Sahara
			556 = { add_contested_owner  = P35 } #Bamako
			776 = { add_contested_owner  = P35 } #Dahomey
			778 = { add_contested_owner  = P35 } #Upper Volta
			779 = { add_contested_owner  = P35 } #Ivory Coast
			780 = { add_contested_owner  = P35 } #Guinea
			781 = { add_contested_owner  = P35 } #Niger
			782 = { add_contested_owner  = P35 } #Tombouctou
			898 = { add_contested_owner  = P35 } #Gao
			899 = { add_contested_owner  = P35 } #Kayes-Koulikoro
			557 = { add_contested_owner  = P35 } #Mauritania
			786 = { add_contested_owner  = P35 } #Mauritanian Desert

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P35 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P35
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P35
				}
			}
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			272 = { add_core_of = P35 } #Senegal
			515 = { add_core_of = P35 } #Southern Sahara
			556 = { add_core_of = P35 } #Bamako
			776 = { add_core_of = P35 } #Dahomey
			778 = { add_core_of = P35 } #Upper Volta
			779 = { add_core_of = P35 } #Ivory Coast
			780 = { add_core_of = P35 } #Guinea
			781 = { add_core_of = P35 } #Niger
			782 = { add_core_of = P35 } #Tombouctou
			898 = { add_core_of = P35 } #Gao
			899 = { add_core_of = P35 } #Kayes-Koulikoro
			557 = { add_core_of = P35 } #Mauritania
			786 = { add_core_of = P35 } #Mauritanian Desert

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P35 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P35
					transfer_state_to = P35
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P35
					set_state_controller_to = P35
				}
			}
		}

		#Transfer all technology from ROOT to formable puppet
		P35 = { pfp_transfer_technology = yes }

		#Set global flag to prevent future decision use
		set_global_flag = pfp_west_africa_flag

		#Send event to all nations about the formable puppet
		set_temp_variable = { temp_country_tag = P35 }
	}
}
