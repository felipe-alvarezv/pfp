#Scripted effect to release Shikoku (Q06)
pfp_release_shikoku = {
    hidden_effect = {
        #Add cores to Shikoku
        530 = { add_core_of = Q06 } #Shikoku
        
        #Set formable puppet as owner and controller of all its cored states
        every_state = {
            limit = { is_core_of = Q06 }
            add_contested_owner = Q06
            transfer_state_to = Q06
        }

        #Transfer all technology to formable puppet
        Q06 = { pfp_transfer_technology = yes }

        #Set global flag to prevent future decision use
        set_global_flag = pfp_shikoku_flag

        #Send event to all nations about the formable puppet
        set_temp_variable = { temp_country_tag = Q06 }
    }
}
