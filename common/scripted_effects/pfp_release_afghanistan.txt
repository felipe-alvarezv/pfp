#Scripted effect to release Afghanistan (P76)
pfp_release_afghanistan = {
	hidden_effect = {
		#Add cores to Afghanistan
		267 = { add_core_of = P76 } #Kabul
		415 = { add_core_of = P76 } #Herat
		1005 = { add_core_of = P76 } #Qataghan
		1006 = { add_core_of = P76 } #Khyber Pass
		1007 = { add_core_of = P76 } #Maymanah
		1008 = { add_core_of = P76 } #Qandahar
		1009 = { add_core_of = P76 } #Farah

		#Set formable puppet as owner and controller of all its cored states
		every_state = {
			limit = { is_core_of = P76 }
			add_contested_owner = P76
			transfer_state_to = P76
		}

		#Transfer all technology to formable puppet
		P76 = { transfer_technology = yes }

		#Set global flag to prevent future decision use
		set_global_flag = pfp_afghanistan_flag

		#Send event to all nations about the formable puppet
		set_temp_variable = { temp_country_tag = P76 }
	}
}
