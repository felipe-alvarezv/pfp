#Scripted effect to release Prussia (P33)
pfp_release_prussia = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_enabled }

			#Add constestion to formable puppet
			5 = { add_contested_owner  = P33 } #Ermland-Masuren
			62 = { add_contested_owner  = P33 } #Vorpommern
			63 = { add_contested_owner  = P33 } #Hinterpommern
			64 = { add_contested_owner  = P33 } #Brandenburg
			66 = { add_contested_owner  = P33 } #Niederschlesien
			67 = { add_contested_owner  = P33 } #Oberschlesien
			68 = { add_contested_owner  = P33 } #Ostmark
			763 = { add_contested_owner  = P33 } #Konigsberg

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P33 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P33
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P33
				}
			}

			#Set custom focus tree to combat resistance
			P33 = { set_focus_tree = pfp_formable_puppet_focus }
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			5 = { add_core_of = P33 } #Ermland-Masuren
			62 = { add_core_of = P33 } #Vorpommern
			63 = { add_core_of = P33 } #Hinterpommern
			64 = { add_core_of = P33 } #Brandenburg
			66 = { add_core_of = P33 } #Niederschlesien
			67 = { add_core_of = P33 } #Oberschlesien
			68 = { add_core_of = P33 } #Ostmark
			763 = { add_core_of = P33 } #Konigsberg

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P33 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P33
					transfer_state_to = P33
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P33
					set_state_controller_to = P33
				}
			}
		}	}
}
