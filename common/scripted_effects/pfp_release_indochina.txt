#Scripted effect to release Indochina (P16)
pfp_release_indochina = {
	hidden_effect = {
		#Add claims to Indochina
		286 = { add_claim_by = P16 } #Southern Indochina
		670 = { add_claim_by = P16 } #Laos
		671 = { add_claim_by = P16 } #Tonkin
		741 = { add_claim_by = P16 } #Cambodia

		#Set formable puppet as owner and controller of all its claimed states
		every_state = {
			limit = { is_claimed_by = P16 }
			add_contested_owner = P16
			transfer_state_to = P16
		}

		#Transfer all technology to formable puppet
		P16 = { transfer_technology = yes }

		#Set global flag to prevent future decision use
		set_global_flag = pfp_indochina_flag

		#Send event to all nations about the formable puppet
		set_temp_variable = { temp_country_tag = P16 }
	}
}
