#Scripted effect to release Indonesia (P17)
pfp_release_indonesia = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_enabled }

			#Add constestion to formable puppet
			334 = { add_contested_owner  = P17 } #Kalimantan
			335 = { add_contested_owner  = P17 } #Java
			667 = { add_contested_owner  = P17 } #Lesser Sunda Islands
			668 = { add_contested_owner  = P17 } #The Moluccas
			672 = { add_contested_owner  = P17 } #Sumatra
			673 = { add_contested_owner  = P17 } #Sulawesi
			1022 = { add_contested_owner  = P17 } #Inner Borneo

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P17 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P17
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P17
				}
			}

			#Set custom focus tree to combat resistance
			P17 = { load_focus_tree = pfp_formable_puppet_focus }
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			334 = { add_core_of = P17 } #Kalimantan
			335 = { add_core_of = P17 } #Java
			667 = { add_core_of = P17 } #Lesser Sunda Islands
			668 = { add_core_of = P17 } #The Moluccas
			672 = { add_core_of = P17 } #Sumatra
			673 = { add_core_of = P17 } #Sulawesi
			1022 = { add_core_of = P17 } #Inner Borneo

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P17 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P17
					transfer_state_to = P17
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P17
					set_state_controller_to = P17
				}
			}
		}	}
}
