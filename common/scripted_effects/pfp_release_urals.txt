#Scripted effect to release Urals (P44)
pfp_release_urals = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_on }

			#Add constestion to formable puppet
			403 = { add_contested_owner  = P44 } #Tyumen
			572 = { add_contested_owner  = P44 } #Chelyabinsk
			573 = { add_contested_owner  = P44 } #Zlatoust
			577 = { add_contested_owner  = P44 } #Surgut
			579 = { add_contested_owner  = P44 } #Salekhard
			580 = { add_contested_owner  = P44 } #Tobolsk
			581 = { add_contested_owner  = P44 } #Northern Urals
			582 = { add_contested_owner  = P44 } #Magnitogorsk
			653 = { add_contested_owner  = P44 } #Sverdlovsk
			824 = { add_contested_owner  = P44 } #Yamalia

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P44 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P44
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P44
				}
			}
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			403 = { add_core_of = P44 } #Tyumen
			572 = { add_core_of = P44 } #Chelyabinsk
			573 = { add_core_of = P44 } #Zlatoust
			577 = { add_core_of = P44 } #Surgut
			579 = { add_core_of = P44 } #Salekhard
			580 = { add_core_of = P44 } #Tobolsk
			581 = { add_core_of = P44 } #Northern Urals
			582 = { add_core_of = P44 } #Magnitogorsk
			653 = { add_core_of = P44 } #Sverdlovsk
			824 = { add_core_of = P44 } #Yamalia

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P44 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P44
					transfer_state_to = P44
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P44
					set_state_controller_to = P44
				}
			}
		}

		#Transfer all technology from ROOT to formable puppet
		P44 = { pfp_transfer_technology = yes }

		#Set global flag to prevent future decision use
		set_global_flag = pfp_urals_flag

		#Send event to all nations about the formable puppet
		set_temp_variable = { temp_country_tag = P44 }
	}
}
