#Scripted effect to release Eastern Canada (P11)
pfp_release_eastern_canada = {
	hidden_effect = {
		#Add claims to Eastern Canada
		276 = { add_claim_by = P11 } #Southern Ontario
		682 = { add_claim_by = P11 } #Northern Ontario
		866 = { add_claim_by = P11 } #Districts of Ontario

		#Set formable puppet as owner and controller of all its claimed states
		every_state = {
			limit = { is_claimed_by = P11 }
			transfer_state_to = P11
		}

		#Transfer all technology to formable puppet
		P11 = { transfer_technology = yes }

		#Set global flag to prevent future decision use
		set_global_flag = pfp_eastern_canada_flag

		#Send event to all nations about the formable puppet
		set_temp_variable = { temp_country_tag = P11 }
		news_event = { id = pfp_news.01 }
	}
}
