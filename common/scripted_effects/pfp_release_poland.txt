#Scripted effect to release Poland (P34)
pfp_release_poland = {
	hidden_effect = {
		#Add claims to Poland
		10 = { add_claim_by = P34 } #Warszawa
		87 = { add_claim_by = P34 } #Lodz
		90 = { add_claim_by = P34 } #Kielce
		92 = { add_claim_by = P34 } #Lublin
		98 = { add_claim_by = P34 } #PÅ‚ock

		#Set formable puppet as owner and controller of all its claimed states
		every_state = {
			limit = { is_claimed_by = P34 }
			transfer_state_to = P34
		}

		#Transfer all technology to formable puppet
		P34 = { transfer_technology = yes }

		#Set global flag to prevent future decision use
		set_global_flag = pfp_poland_flag

		#Send event to all nations about the formable puppet
		set_temp_variable = { temp_country_tag = P34 }
		news_event = { id = pfp_news.01 }
	}
}
