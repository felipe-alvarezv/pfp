#Scripted Effect to set "Military Occupation" color
pfp_set_autonomy_color = {
    if = {
        limit = { has_global_flag = pfp_use_overlord_color } #Check if color decision is toggled on
        var:pfp_portrait_tag  = {
            set_temp_variable = { target_military_occupation = THIS }
            overlord = { set_autonomy = { target = target_military_occupation autonomy_state = pfp_autonomy_military_occupation_color } } #Toggle overlord color on
        }
    }
    else = {
        var:pfp_portrait_tag  = {
            set_temp_variable = { target_military_occupation = THIS }
            overlord = { set_autonomy = { target = target_military_occupation autonomy_state = pfp_autonomy_military_occupation } } #Toggle overlord color off
        }
    }
}
