#Scripted effect to release Tibet (Q24)
pfp_release_tibet = {
    hidden_effect = {
        #Add cores to Tibet
        322 = { add_core_of = Q24 } #Nagqu
        601 = { add_core_of = Q24 } #Kham
        757 = { add_core_of = Q24 } #Shigatse
        758 = { add_core_of = Q24 } #Ngari
        
        #Set formable puppet as owner and controller of all its cored states
        every_state = {
            limit = { is_core_of = Q24 }
            add_contested_owner = Q24
            transfer_state_to = Q24
        }

        #Transfer all technology to formable puppet
        Q24 = { pfp_transfer_technology = yes }

        #Set global flag to prevent future decision use
        set_global_flag = pfp_tibet_flag

        #Send event to all nations about the formable puppet
        set_temp_variable = { temp_country_tag = Q24 }
    }
}
