#Scripted effects to release North Africa (P30)
pfp_release_north_africa = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_enabled }

			#Add constestion to formable puppet
			273 = { add_contested_owner  = P30 } #Lybian Desert
			447 = { add_contested_owner  = P30 } #Alexandria
			448 = { add_contested_owner  = P30 } #Tripoli
			449 = { add_contested_owner  = P30 } #El Agheila
			450 = { add_contested_owner  = P30 } #Benghasi
			451 = { add_contested_owner  = P30 } #Derna
			452 = { add_contested_owner  = P30 } #Matrouh
			456 = { add_contested_owner  = P30 } #Aswan
			457 = { add_contested_owner  = P30 } #Eastern Desert
			552 = { add_contested_owner  = P30 } #Western Desert
			661 = { add_contested_owner  = P30 } #Tripolitania
			662 = { add_contested_owner  = P30 } #Sirte
			663 = { add_contested_owner  = P30 } #Cyrenaica
			907 = { add_contested_owner  = P30 } #Cairo

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P30 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P30
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P30
				}
			}

			#Set custom focus tree to combat resistance
			P30 = { set_focus_tree = pfp_formable_puppet_focus }
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			273 = { add_core_of = P30 } #Lybian Desert
			447 = { add_core_of = P30 } #Alexandria
			448 = { add_core_of = P30 } #Tripoli
			449 = { add_core_of = P30 } #El Agheila
			450 = { add_core_of = P30 } #Benghasi
			451 = { add_core_of = P30 } #Derna
			452 = { add_core_of = P30 } #Matrouh
			456 = { add_core_of = P30 } #Aswan
			457 = { add_core_of = P30 } #Eastern Desert
			552 = { add_core_of = P30 } #Western Desert
			661 = { add_core_of = P30 } #Tripolitania
			662 = { add_core_of = P30 } #Sirte
			663 = { add_core_of = P30 } #Cyrenaica
			907 = { add_core_of = P30 } #Cairo

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P30 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P30
					transfer_state_to = P30
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P30
					set_state_controller_to = P30
				}
			}
		}	}
}
