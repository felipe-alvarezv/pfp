#Scripted effect to release Pakistan (P75)
pfp_release_pakistan = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_on }

			#Add constestion to formable puppet
			443 = { add_contested_owner  = P75 } #Sind
			444 = { add_contested_owner  = P75 } #South Baluchistan
			445 = { add_contested_owner  = P75 } #Sibi
			987 = { add_contested_owner  = P75 } #Wazaristan
			988 = { add_contested_owner  = P75 } #North Baluchistan
			1012 = { add_contested_owner  = P75 } #Kalat

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P75 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P75
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P75
				}
			}
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			443 = { add_core_of = P75 } #Sind
			444 = { add_core_of = P75 } #South Baluchistan
			445 = { add_core_of = P75 } #Sibi
			987 = { add_core_of = P75 } #Wazaristan
			988 = { add_core_of = P75 } #North Baluchistan
			1012 = { add_core_of = P75 } #Kalat

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P75 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P75
					transfer_state_to = P75
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P75
					set_state_controller_to = P75
				}
			}
		}

		#Transfer all technology to formable puppet
		P75 = { pfp_transfer_technology = yes }

		#Set global flag to prevent future decision use
		set_global_flag = pfp_pakistan_flag

		#Send event to all nations about the formable puppet
		set_temp_variable = { temp_country_tag = P75 }
	}
}
