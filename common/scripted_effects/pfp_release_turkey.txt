#Scripted effect to release Turkey (P68)
pfp_release_turkey = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_on }

			#Add constestion to formable puppet
			49 = { add_contested_owner  = P68 } #Ankara
			343 = { add_contested_owner  = P68 } #Afyon
			346 = { add_contested_owner  = P68 } #Konya
			348 = { add_contested_owner  = P68 } #Kayseri
			349 = { add_contested_owner  = P68 } #Sivas
			355 = { add_contested_owner  = P68 } #Samsun
			356 = { add_contested_owner  = P68 } #Kastamonu
			798 = { add_contested_owner  = P68 } #Amasya

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P68 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P68
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P68
				}
			}
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			49 = { add_core_of = P68 } #Ankara
			343 = { add_core_of = P68 } #Afyon
			346 = { add_core_of = P68 } #Konya
			348 = { add_core_of = P68 } #Kayseri
			349 = { add_core_of = P68 } #Sivas
			355 = { add_core_of = P68 } #Samsun
			356 = { add_core_of = P68 } #Kastamonu
			798 = { add_core_of = P68 } #Amasya

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P68 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P68
					transfer_state_to = P68
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P68
					set_state_controller_to = P68
				}
			}
		}

		#Transfer all technology from ROOT to formable puppet
		P68 = { pfp_transfer_technology = yes }

		#Set global flag to prevent future decision use
		set_global_flag = pfp_turkey_flag

		#Send event to all nations about the formable puppet
		set_temp_variable = { temp_country_tag = P68 }
	}
}
